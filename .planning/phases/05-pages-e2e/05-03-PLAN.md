```ts
import { test, expect } from '@playwright/test';

/**
 * E2E #1 — Apply flow happy path (TST-06).
 *
 * Visits /today, mocks the API surface so a deterministic Top-5 card is present
 * with a known URL, intercepts POST /api/actions/apply to return 200, clicks the
 * card's [Open] button, asserts a success indication in the UI.
 *
 * NOTE: ListingCard.onOpen on /today is currently not wired to /api/actions/apply
 * (that happens inside the ListingModal). For the /today landing, [Open] is a no-op
 * by default in the v1 Plan 05-01 contract. This spec therefore drives the flow
 * through the MODAL: it opens /pipeline, clicks the first row, then [Open in Chrome]
 * in the modal. That exercises the same /api/actions/apply endpoint with full UX.
 */
test.describe('apply-flow', () => {
  test('opens a listing in Chrome via the modal and shows success', async ({ page }) => {
    // 1. Mock pipeline so we have a deterministic clickable row.
    await page.route('**/api/pipeline', async (route) => {
      await route.fulfill({
        status: 200,
        contentType: 'application/json',
        body: JSON.stringify({
          data: [
            {
              state: 'evaluated',
              num: 42,
              url: 'https://example.com/jobs/42',
              company: 'Acme Robotics',
              title: 'Senior Designer',
              score: 4.5,
              pdf: true,
              note: null,
            },
          ],
          errors: [],
        }),
      });
    });

    // 2. Mock listing fetch so the modal renders.
    await page.route('**/api/listing/42', async (route) => {
      await route.fulfill({
        status: 200,
        contentType: 'application/json',
        body: JSON.stringify({
          report: {
            num: 42,
            slug: 'acme-robotics',
            date: '2026-05-20',
            title: 'Senior Designer at Acme Robotics',
            score: 4.5,
            url: 'https://example.com/jobs/42',
            pdf: true,
            legitimacy: 'verified',
            blocks: {
              A: { score: 4.5, notes: 'Strong fit' },
              B: null, C: null, D: null, E: null, F: null,
            },
            body: '# Acme Robotics\nFit details.',
          },
          pdfPath: null,
        }),
      });
    });

    // 3. Intercept the apply POST and assert payload shape.
    let applyCalled = false;
    let applyPayload: unknown = null;
    await page.route('**/api/actions/apply', async (route) => {
      applyCalled = true;
      const req = route.request();
      try { applyPayload = JSON.parse(req.postData() ?? '{}'); } catch { applyPayload = null; }
      await route.fulfill({
        status: 200,
        contentType: 'application/json',
        body: JSON.stringify({ ok: true }),
      });
    });

    // 4. Drive the UI.
    await page.goto('/pipeline');
    await expect(page.getByTestId('pipeline-table-root')).toBeVisible();
    await page.getByTestId('pipeline-row-42').click();

    const modal = page.getByTestId('listing-modal');
    await expect(modal).toBeVisible();
    await expect(modal.getByTestId('modal-md-pane')).toContainText('Acme Robotics');

    await modal.getByTestId('modal-action-open').click();

    // 5. Assert /api/actions/apply was hit with the right URL, and UI shows success.
    await expect.poll(() => applyCalled, { timeout: 5_000 }).toBe(true);
    expect(applyPayload).toMatchObject({ url: 'https://example.com/jobs/42' });

    await expect(modal.getByTestId('modal-apply-message')).toHaveText(/opened in Chrome/i);
  });
});
```

Notes:
- We route /pipeline through `page.route` instead of `/today` because the [Open] flow lives inside the ListingModal which is opened from /pipeline rows. The /today [Open] buttons (on ListingCard) are decorative in v1 and do not POST to /api/actions/apply per Plan 05-01.
- `expect.poll` is used to confirm the mock fired without a brittle wait.
- The PDF iframe is intentionally NOT loaded (we set pdfPath: null) to avoid Playwright cross-frame complications in the happy spec.

```ts
import { test, expect } from '@playwright/test';

/**
 * E2E #2 — Mark-sent lock contention (TST-06).
 *
 * Mocks /api/actions/mark-sent to return 423 Locked and asserts the modal
 * surfaces "Locked, try again" inline (modal MUST stay open).
 */
test.describe('mark-sent-lock', () => {
  test('shows "Locked, try again" inline when mark-sent returns 423', async ({ page }) => {
    // 1. Mock pipeline with a deterministic row.
    await page.route('**/api/pipeline', async (route) => {
      await route.fulfill({
        status: 200,
        contentType: 'application/json',
        body: JSON.stringify({
          data: [
            {
              state: 'evaluated',
              num: 17,
              url: 'https://example.com/jobs/17',
              company: 'LockCorp',
              title: 'Lead Engineer',
              score: 4.2,
              pdf: false,
              note: null,
            },
          ],
          errors: [],
        }),
      });
    });

    // 2. Mock listing fetch.
    await page.route('**/api/listing/17', async (route) => {
      await route.fulfill({
        status: 200,
        contentType: 'application/json',
        body: JSON.stringify({
          report: {
            num: 17,
            slug: 'lockcorp',
            date: '2026-05-20',
            title: 'Lead Engineer at LockCorp',
            score: 4.2,
            url: 'https://example.com/jobs/17',
            pdf: false,
            legitimacy: null,
            blocks: { A: null, B: null, C: null, D: null, E: null, F: null },
            body: 'Body.',
          },
          pdfPath: null,
        }),
      });
    });

    // 3. Mock mark-sent → 423.
    let markCalled = false;
    let markPayload: unknown = null;
    await page.route('**/api/actions/mark-sent', async (route) => {
      markCalled = true;
      try { markPayload = JSON.parse(route.request().postData() ?? '{}'); } catch { markPayload = null; }
      await route.fulfill({
        status: 423,
        contentType: 'application/json',
        body: JSON.stringify({ error: 'Locked' }),
      });
    });

    // 4. Drive UI.
    await page.goto('/pipeline');
    await page.getByTestId('pipeline-row-17').click();

    const modal = page.getByTestId('listing-modal');
    await expect(modal).toBeVisible();
    await modal.getByTestId('modal-action-mark-applied').click();

    // 5. Assert request was made AND modal stays open with locked message.
    await expect.poll(() => markCalled, { timeout: 5_000 }).toBe(true);
    expect(markPayload).toMatchObject({ id: '17', status: 'Applied' });

    await expect(modal.getByTestId('modal-mark-message')).toHaveText(/Locked, try again/i);
    await expect(modal).toBeVisible(); // Still mounted — did not auto-close.
  });
});
```

Notes:
- The test asserts `modal-mark-message` (not `modal-apply-message`) — those are two different feedback channels per Plan 05-02 modal contract.
- After 423, the modal MUST NOT close. The spec explicitly re-asserts `await expect(modal).toBeVisible()` after the failure.
- payload assertion confirms the wrapper builds the right body: `{ id, status: 'Applied' }`. The `id` is the report.num as a string (per Plan 03-02 contract).

```ts
import { test, expect } from '@playwright/test';

/**
 * E2E #3 — Malformed MD error boundary (TST-06).
 *
 * Mocks /api/applications to return a payload with errors[].length > 0,
 * visits /today, asserts the parse-errors-banner is visible, and asserts the
 * page does NOT escalate to a Next.js error boundary (Today. hero still
 * renders, route did not crash).
 *
 * Since /today is a server component that calls parseApplications directly,
 * the API route mock would NOT influence the server render. We instead validate
 * the same UX path by:
 *   1. Hitting the API endpoint directly to confirm the parser emits errors[]
 *      (using the fixtures committed in 3a/3b — wired via NEXT_PUBLIC_TEST_FIXTURES
 *      env override OR an injection point in api-paths).
 *   2. Mocking the api-paths layer is out of scope for E2E; instead we
 *      mock the network at the browser fetch level — and accept the fact
 *      that for SSR pages the banner is rendered by the server.
 *
 * Approach taken: drive the assertion via a CLIENT-side test surface. We don't
 * need to mock at SSR level — the banner is rendered by `<TodayHero parseErrors={...}/>`
 * which receives errors directly from the server component. To inject errors
 * deterministically without touching the filesystem, we use a test-only query
 * parameter `?e2e_inject_errors=N` that the page reads (added in fallback path
 * below) — OR we use route.fulfill on `/api/applications` and click a hypothetical
 * "refresh from API" button — but neither exists in v1.
 *
 * SIMPLEST WORKING APPROACH: write the malformed fixture to data/applications.md
 * BEFORE goto, restore after the test. This drives the real server render.
 *
 * That's brittle if anyone else uses the repo at the same time, so we use
 * a Playwright BROWSER context route to swap the file path: NOT possible.
 *
 * FINAL APPROACH: temporarily rename data/applications.md to .bak, copy our
 * fixture into place, run the assertion, restore.
 */

import { promises as fs } from 'node:fs';
import path from 'node:path';

const REPO_ROOT = path.resolve(__dirname, '../../..');
const APPLICATIONS_REAL = path.join(REPO_ROOT, 'data/applications.md');
const APPLICATIONS_BACKUP = path.join(REPO_ROOT, 'data/applications.md.e2e-bak');
const FIXTURE_PATH = path.join(__dirname, 'fixtures/applications-malformed.md');

test.describe('malformed-md', () => {
  test.beforeAll(async () => {
    // Snapshot the real file and substitute the malformed fixture.
    try {
      await fs.copyFile(APPLICATIONS_REAL, APPLICATIONS_BACKUP);
      const fixture = await fs.readFile(FIXTURE_PATH, 'utf8');
      await fs.writeFile(APPLICATIONS_REAL, fixture, 'utf8');
    } catch (e) {
      throw new Error(`malformed-md setup failed: ${e instanceof Error ? e.message : String(e)}`);
    }
  });

  test.afterAll(async () => {
    // Restore the snapshot, even if the test failed.
    try {
      await fs.copyFile(APPLICATIONS_BACKUP, APPLICATIONS_REAL);
      await fs.unlink(APPLICATIONS_BACKUP);
    } catch (e) {
      // Surface but don't throw — test result must not depend on cleanup.
      console.error('malformed-md cleanup warning:', e);
    }
  });

  test('shows parse-errors-banner without crashing', async ({ page }) => {
    await page.goto('/');

    // Today. heading must still render (page did not crash).
    await expect(page.getByTestId('today-hero-heading')).toHaveText(/Today\./);

    // The parse-errors banner is visible.
    const banner = page.getByTestId('parse-errors-banner');
    await expect(banner).toBeVisible();
    await expect(banner).toContainText(/parse error/i);

    // No Next.js error boundary signature visible.
    await expect(page.locator('text=Application error: a server-side exception')).toHaveCount(0);
  });
});
```

Notes for executor:
- The snapshot-restore-around-test pattern is the cleanest way to drive a server-rendered /today through a malformed-MD scenario without mocking the filesystem layer.
- `playwright.config.ts` has `workers: 1` (serial execution) — guaranteed by Phase 1 — so the file-swap is safe.
- The `afterAll` restore runs even on test failure (Playwright behavior). If the test process is SIGKILLed mid-test the backup remains as `data/applications.md.e2e-bak` — that's recoverable manually.
- The path math `path.resolve(__dirname, '../../..')` assumes the spec runs from `dashboard/web/e2e/`. Verify with `console.log(REPO_ROOT)` if the first run fails.
